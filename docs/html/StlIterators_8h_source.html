56 template<typename XprType>
57 class pointer_based_stl_iterator
58 {
59  enum { is_lvalue = internal::is_lvalue<XprType>::value };
60 public:
61  typedef Index difference_type;
62  typedef typename XprType::Scalar value_type;
63  typedef std::random_access_iterator_tag iterator_category;
64  typedef typename internal::conditional<bool(is_lvalue), value_type*, const value_type*>::type pointer;
65  typedef typename internal::conditional<bool(is_lvalue), value_type&, const value_type&>::type reference;
66 
67  pointer_based_stl_iterator() : m_ptr(0) {}
68  pointer_based_stl_iterator(XprType& xpr, Index index) : m_incr(xpr.innerStride())
69  {
70  m_ptr = xpr.data() + index * m_incr.value();
71  }
72 
73  reference operator*() const { return *m_ptr; }
74  reference operator[](Index i) const { return *(m_ptr+i*m_incr.value()); }
75  pointer operator->() const { return m_ptr; }
76 
77  pointer_based_stl_iterator& operator++() { m_ptr += m_incr.value(); return *this; }
78  pointer_based_stl_iterator& operator--() { m_ptr -= m_incr.value(); return *this; }
79 
80  pointer_based_stl_iterator operator++(int) { pointer_based_stl_iterator prev(*this); operator++(); return prev;}
81  pointer_based_stl_iterator operator--(int) { pointer_based_stl_iterator prev(*this); operator--(); return prev;}
82 
83  friend pointer_based_stl_iterator operator+(const pointer_based_stl_iterator& a, Index b) { pointer_based_stl_iterator ret(a); ret += b; return ret; }
84  friend pointer_based_stl_iterator operator-(const pointer_based_stl_iterator& a, Index b) { pointer_based_stl_iterator ret(a); ret -= b; return ret; }
85  friend pointer_based_stl_iterator operator+(Index a, const pointer_based_stl_iterator& b) { pointer_based_stl_iterator ret(b); ret += a; return ret; }
86  friend pointer_based_stl_iterator operator-(Index a, const pointer_based_stl_iterator& b) { pointer_based_stl_iterator ret(b); ret -= a; return ret; }
87 
88  pointer_based_stl_iterator& operator+=(Index b) { m_ptr += b*m_incr.value(); return *this; }
89  pointer_based_stl_iterator& operator-=(Index b) { m_ptr -= b*m_incr.value(); return *this; }
90 
91  difference_type operator-(const pointer_based_stl_iterator& other) const {
92  return (m_ptr - other.m_ptr)/m_incr.value();
93  }
94 
95  bool operator==(const pointer_based_stl_iterator& other) { return m_ptr == other.m_ptr; }
96  bool operator!=(const pointer_based_stl_iterator& other) { return m_ptr != other.m_ptr; }
97  bool operator< (const pointer_based_stl_iterator& other) { return m_ptr < other.m_ptr; }
98  bool operator<=(const pointer_based_stl_iterator& other) { return m_ptr <= other.m_ptr; }
99  bool operator> (const pointer_based_stl_iterator& other) { return m_ptr > other.m_ptr; }
100  bool operator>=(const pointer_based_stl_iterator& other) { return m_ptr >= other.m_ptr; }
101 
102 protected:
103 
104  pointer m_ptr;
105  internal::variable_if_dynamic<Index, XprType::InnerStrideAtCompileTime> m_incr;
106 };
