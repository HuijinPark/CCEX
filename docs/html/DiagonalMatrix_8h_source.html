314 template<typename Derived>
315 bool MatrixBase<Derived>::isDiagonal(const RealScalar& prec) const
316 {
317  if(cols() != rows()) return false;
318  RealScalar maxAbsOnDiagonal = static_cast<RealScalar>(-1);
319  for(Index j = 0; j < cols(); ++j)
320  {
321  RealScalar absOnDiagonal = numext::abs(coeff(j,j));
322  if(absOnDiagonal > maxAbsOnDiagonal) maxAbsOnDiagonal = absOnDiagonal;
323  }
324  for(Index j = 0; j < cols(); ++j)
325  for(Index i = 0; i < j; ++i)
326  {
327  if(!internal::isMuchSmallerThan(coeff(i, j), maxAbsOnDiagonal, prec)) return false;
328  if(!internal::isMuchSmallerThan(coeff(j, i), maxAbsOnDiagonal, prec)) return false;
329  }
330  return true;
331 }
